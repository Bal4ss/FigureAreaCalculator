<a name="a3"><h3>Ответ 3</h3></a>
> SELECT Product.Name, Category.Name
>	FROM Product
>	LEFT JOIN ProductCategory ON Product.Id = ProductCategory.ProductId
>	LEFT JOIN Category ON Category.Id = ProductCategory.CategoryId
